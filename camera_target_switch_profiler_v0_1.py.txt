def normalized(value) -> str:
    return re.sub(r"[^a-z0-9]", "", str(value).lower())

    try:

def sniff_csv(path: Path):
    raw = path.read_bytes()[:131072]
    decoded = None
    encoding = None
    for candidate in ["utf-8-sig", "cp932", "shift_jis", "utf-8", "latin1"]:
        try:
            decoded = raw.decode(candidate)
            encoding = candidate
            break
        except UnicodeDecodeError:
            continue
    if decoded is None:
        decoded = raw.decode("latin1", errors="replace")
        encoding = "latin1"
    try:
        separator = csv.Sniffer().sniff(decoded, delimiters=",;\t|").delimiter
    except Exception:
        counts = {item: decoded.count(item) for item in [",", ";", "\t", "|"]}
        separator = max(counts, key=counts.get)
    return encoding, separator


def read_header(path: Path):
    import pandas as pd

    if path.suffix.lower() in {".xlsx", ".xls", ".xlsm"}:
        frame = pd.read_excel(path, nrows=0)
        return list(frame.columns), {"kind": "excel"}
    encoding, separator = sniff_csv(path)
    frame = pd.read_csv(path, sep=separator, encoding=encoding, nrows=0)
    return list(frame.columns), {
        "kind": "csv",
        "encoding": encoding,
        "separator": separator,
    }


def iter_table(path: Path, reader_info: dict, usecols):
    import pandas as pd

    if reader_info["kind"] == "excel":
        yield pd.read_excel(path, usecols=usecols)
        return
    yield from pd.read_csv(
        path,
        sep=reader_info["separator"],
        encoding=reader_info["encoding"],
        usecols=usecols,
        chunksize=CHUNK_SIZE,
        low_memory=False,
    )


def detect_framecount(columns):
    exact = [column for column in columns if normalized(column) == "framecount"]
    named = [column for column in columns if "framecount" in normalized(column)]
    id620 = [
        column
        for column in columns
        if "620" in normalized(column)
        and any(token in normalized(column) for token in ["frame", "count"])
    ]
    return (exact or named or id620 or [None])[0]


def parse_sentinels(text: str):
    values = set()
    for item in re.split(r"[,;\s]+", text.strip()):
        if not item:
            continue
        try:
            values.add(float(item))
        except ValueError:
            pass
    return values

def fast_fingerprint(path: Path):
    digest = hashlib.sha256()
    size = path.stat().st_size
    digest.update(str(size).encode("ascii"))
    with path.open("rb") as handle:
        digest.update(handle.read(1024 * 1024))
        if size > 1024 * 1024:
            handle.seek(max(0, size - 1024 * 1024))
            digest.update(handle.read(1024 * 1024))
    return digest.hexdigest()


def is_valid_number(value, sentinels):
    if value is None or not math.isfinite(value):
        return False
    return not any(math.isclose(value, sentinel, rel_tol=1e-12, abs_tol=1e-12) for sentinel in sentinels)


def add_value(record, key, value, sentinels=()):
    if value is None or not math.isfinite(value):
        return
    if key.endswith("distance") and not is_valid_number(value, sentinels):
        return
    record[key].add(float(value))


def unique_value(record, key):
    values = record.get(key, set())
    if len(values) == 1:
        return next(iter(values)), False
    if len(values) > 1:
        return None, True
    return None, False


def rate_code(rate):
    if rate == 0:
        return 1
    if rate < 0.10:
        return 2
    if rate < 0.90:
        return 3
    return 4


def count_code(count):
    if count == 0:
        return 1
    if count <= 10:
        return 2
    return 3


def profile(path: Path, config: dict, output_dir: Path):
    import pandas as pd

    output_dir.mkdir(parents=True, exist_ok=True)
    columns, reader_info = read_header(path)
    selected_mapping = {
        "frame": config["frame"],
        "radar_distance": config["radar_distance"],
        "cam_a_distance": config["cam_a_distance"],
        "cam_b_distance": config["cam_b_distance"],
        "reference_id": config.get("reference_id"),
        "cam_a_id": config.get("cam_a_id"),
        "cam_b_id": config.get("cam_b_id"),
    }
    usecols = list(dict.fromkeys(value for value in selected_mapping.values() if value))
    before_stat = path.stat()
    started_perf = time.perf_counter()

    frames = defaultdict(lambda: defaultdict(set))
    segment_id = 0
    previous_input_frame = None
    total_rows = 0
    backward_transitions = 0

    for chunk_index, chunk in enumerate(iter_table(path, reader_info, usecols), start=1):
        numeric = {
            role: pd.to_numeric(chunk[column], errors="coerce")
            for role, column in selected_mapping.items()
            if column
        }
        row_count = len(chunk)
        total_rows += row_count

        arrays = {role: series.to_numpy() for role, series in numeric.items()}
        for row_index in range(row_count):
            frame_value = arrays["frame"][row_index]
            if pd.isna(frame_value):
                continue
            frame_value = float(frame_value)
            if previous_input_frame is not None and frame_value < previous_input_frame:
                segment_id += 1
                backward_transitions += 1
            previous_input_frame = frame_value
            record = frames[(segment_id, frame_value)]

            add_value(
                record,
                "radar_distance",
                float(arrays["radar_distance"][row_index])
                if not pd.isna(arrays["radar_distance"][row_index])
                else None,
                config["radar_sentinels"],
            )
            add_value(
                record,
                "cam_a_distance",
                float(arrays["cam_a_distance"][row_index])
                if not pd.isna(arrays["cam_a_distance"][row_index])
                else None,
                config["cam_a_sentinels"],
            )
            add_value(
                record,
                "cam_b_distance",
                float(arrays["cam_b_distance"][row_index])
                if not pd.isna(arrays["cam_b_distance"][row_index])
                else None,
                config["cam_b_sentinels"],
            )
            for role in ["reference_id", "cam_a_id", "cam_b_id"]:
                if role in arrays and not pd.isna(arrays[role][row_index]):
                    add_value(record, role, float(arrays[role][row_index]))

        elapsed = max(1e-9, time.perf_counter() - started_perf)
        print(
            f"[TargetProfiler] {total_rows:,} rows | {total_rows / elapsed:,.0f} rows/s",
            file=sys.stderr,
        )

    frame_rows = []
    switch_rows = []
    counts = Counter()
    last_segment = None
    last_selected = None
    last_selected_discrepancy = None
    selected_sequence = []

    for segment_frame in sorted(frames):
        segment, frame = segment_frame
        if last_segment is None or segment != last_segment:
            last_segment = segment
            last_selected = None
            last_selected_discrepancy = None
        record = frames[segment_frame]
        radar, radar_conflict = unique_value(record, "radar_distance")
        cam_a, cam_a_conflict = unique_value(record, "cam_a_distance")
        cam_b, cam_b_conflict = unique_value(record, "cam_b_distance")
        reference_id, reference_id_conflict = unique_value(record, "reference_id")
        cam_a_id, cam_a_id_conflict = unique_value(record, "cam_a_id")
        cam_b_id, cam_b_id_conflict = unique_value(record, "cam_b_id")

        if cam_a is not None:
            counts["cam_a_available"] += 1
        if cam_b is not None:
            counts["cam_b_available"] += 1
        if cam_a is not None and cam_b is not None:
            counts["both_available"] += 1
        if cam_a is None and cam_b is None:
            counts["neither_available"] += 1
        if cam_a_conflict:
            counts["cam_a_conflict"] += 1
        if cam_b_conflict:
            counts["cam_b_conflict"] += 1

        selected = "UNKNOWN"
        selection_reason = "ID_FIELDS_UNAVAILABLE"
        if reference_id is not None and cam_a_id is not None and cam_b_id is not None:
            match_a = math.isclose(reference_id, cam_a_id, rel_tol=0, abs_tol=1e-12)
            match_b = math.isclose(reference_id, cam_b_id, rel_tol=0, abs_tol=1e-12)
            if match_a and not match_b:
                selected = "A"
                selection_reason = "REFERENCE_ID_MATCH_A"
            elif match_b and not match_a:
                selected = "B"
                selection_reason = "REFERENCE_ID_MATCH_B"
            elif match_a and match_b:
                selected = "AMBIGUOUS"
                selection_reason = "REFERENCE_ID_MATCH_BOTH"
            else:
                selected = "UNMATCHED"
                selection_reason = "REFERENCE_ID_MATCH_NONE"

        selected_distance = cam_a if selected == "A" else cam_b if selected == "B" else None
        discrepancy = (
            radar - selected_distance
            if radar is not None and selected_distance is not None
            else None
        )

        if selected in {"A", "B"}:
            counts[f"selected_{selected.lower()}"] += 1
            counts["id_selected"] += 1
            selected_sequence.append((segment, selected))
        elif selected == "UNMATCHED":
            counts["id_unmatched"] += 1
        elif selected == "AMBIGUOUS":
            counts["id_ambiguous"] += 1
        else:
            counts["id_unknown"] += 1

        is_switch = (
            selected in {"A", "B"}
            and last_selected in {"A", "B"}
            and selected != last_selected
        )
        jump = None
        jump_over_threshold = False
        if is_switch:
            counts["switch"] += 1
            if discrepancy is not None and last_selected_discrepancy is not None:
                jump = discrepancy - last_selected_discrepancy
                if abs(jump) >= config["jump_threshold"]:
                    jump_over_threshold = True
                    counts["switch_jump_over_threshold"] += 1
            switch_rows.append(
                {
                    "frame_segment_id": segment,
                    "Framecount": frame,
                    "from_candidate": last_selected,
                    "to_candidate": selected,
                    "selected_discrepancy": discrepancy,
                    "discrepancy_jump": jump,
                    "jump_over_threshold": jump_over_threshold,
                }
            )

        if selected in {"A", "B"}:
            last_selected = selected
            if discrepancy is not None:
                last_selected_discrepancy = discrepancy

        frame_rows.append(
            {
                "frame_segment_id": segment,
                "Framecount": frame,
                "radar_distance_unique": radar,
                "camera_a_distance_unique": cam_a,
                "camera_b_distance_unique": cam_b,
                "reference_id_unique": reference_id,
                "camera_a_id_unique": cam_a_id,
                "camera_b_id_unique": cam_b_id,
                "selected_candidate": selected,
                "selection_reason": selection_reason,
                "selected_camera_distance": selected_distance,
                "selected_discrepancy": discrepancy,
                "camera_a_intra_frame_conflict": cam_a_conflict,
                "camera_b_intra_frame_conflict": cam_b_conflict,
                "id_intra_frame_conflict": (
                    reference_id_conflict or cam_a_id_conflict or cam_b_id_conflict
                ),
                "switch_from_previous_selected": is_switch,
            }
        )

    back_and_forth = 0
    for index in range(2, len(selected_sequence)):
        if (
            selected_sequence[index][0] == selected_sequence[index - 1][0]
            and selected_sequence[index][0] == selected_sequence[index - 2][0]
            and selected_sequence[index][1] == selected_sequence[index - 2][1]
            and selected_sequence[index][1] != selected_sequence[index - 1][1]
        ):
            back_and_forth += 1
    counts["back_and_forth"] = back_and_forth

    total_frames = len(frame_rows)
    id_fields_complete = all(
        config.get(key) for key in ["reference_id", "cam_a_id", "cam_b_id"]
    )
    input_after = path.stat()
    input_unchanged = (
        input_after.st_size == before_stat.st_size
        and input_after.st_mtime_ns == before_stat.st_mtime_ns
    )

    frame_path = output_dir / "camera_target_frame_profile_v0_1.csv"
    switch_path = output_dir / "camera_target_switch_events_v0_1.csv"
    manifest_path = output_dir / "camera_target_switch_manifest_v0_1.json"
    log_path = output_dir / "camera_target_switch_log_v0_1.txt"

    frame_fields = list(frame_rows[0].keys()) if frame_rows else []
    with frame_path.open("w", encoding="utf-8-sig", newline="") as handle:
        if frame_fields:
            writer = csv.DictWriter(handle, fieldnames=frame_fields)
            writer.writeheader()
            writer.writerows(frame_rows)

    switch_fields = [
        "frame_segment_id",
        "Framecount",
        "from_candidate",
        "to_candidate",
        "selected_discrepancy",
        "discrepancy_jump",
        "jump_over_threshold",
    ]
    with switch_path.open("w", encoding="utf-8-sig", newline="") as handle:
        writer = csv.DictWriter(handle, fieldnames=switch_fields)
        writer.writeheader()
        writer.writerows(switch_rows)

    manifest = {
        "version": VERSION,
        "run_id": str(uuid.uuid4()),
        "created_utc": datetime.now(timezone.utc).isoformat(),
        "input": {
            "basename": path.name,
            "size_bytes": before_stat.st_size,
            "fast_fingerprint_sha256": fast_fingerprint(path),
            "input_unchanged": input_unchanged,
        },
        "configuration": {
            key: sorted(value) if isinstance(value, set) else value
            for key, value in config.items()
            if key != "confirmed"
        },
        "scope": {
            "total_input_rows": total_rows,
            "total_frame_records": total_frames,
            "segment_count": segment_id + 1 if total_frames else 0,
            "backward_transition_count": backward_transitions,
        },
        "counts": dict(counts),
        "rates": {
            key: value / max(1, total_frames)
            for key, value in counts.items()
            if key not in {"switch", "back_and_forth", "switch_jump_over_threshold"}
        },
        "interpretation_limits": [
            "Radar is a provisional reference, not ground truth.",
            "The profiler never chooses a camera candidate by distance closeness to Radar.",
            "Without complete ID fields, target selection remains unresolved.",
            "Configured sentinel values are local test assumptions and require business confirmation.",
            "A switch is an engineering review candidate, not root cause.",
        ],
    }
    manifest_path.write_text(
        json.dumps(manifest, ensure_ascii=False, indent=2), encoding="utf-8"
    )

    log_path.write_text(
        "\n".join(
            [
                f"input_basename={path.name}",
                f"total_input_rows={total_rows}",
                f"total_frames={total_frames}",
                f"segment_count={segment_id + 1 if total_frames else 0}",
                f"backward_transitions={backward_transitions}",
                f"id_fields_complete={id_fields_complete}",
                f"counts={dict(counts)}",
                f"elapsed_seconds={time.perf_counter() - started_perf:.3f}",
                f"input_unchanged={input_unchanged}",
            ]
        )
        + "\n",
        encoding="utf-8",
    )

    rate_a = counts["cam_a_available"] / max(1, total_frames)
    rate_b = counts["cam_b_available"] / max(1, total_frames)
    rate_both = counts["both_available"] / max(1, total_frames)
    rate_neither = counts["neither_available"] / max(1, total_frames)
    id_mapped_rate = counts["id_selected"] / max(1, total_frames)
    selected_a_rate = counts["selected_a"] / max(1, counts["id_selected"])
    selected_b_rate = counts["selected_b"] / max(1, counts["id_selected"])

    if total_frames < 100_000:
        frame_bucket = 1
    elif total_frames < 500_000:
        frame_bucket = 2
    else:
        frame_bucket = 3

    id_field_code = 3 if id_fields_complete else (
        1 if config.get("reference_id") else 2 if config.get("cam_a_id") or config.get("cam_b_id") else 0
    )

    codes = [
        2,
        2,
        2,
        2,
        1 if backward_transitions == 0 else 2,
        frame_bucket,
        rate_code(rate_a),
        rate_code(rate_b),
        rate_code(rate_both),
        rate_code(rate_neither),
        count_code(counts["cam_a_conflict"]),
        count_code(counts["cam_b_conflict"]),
        id_field_code,
        rate_code(id_mapped_rate) if id_fields_complete else 0,
        rate_code(selected_a_rate) if counts["id_selected"] else 0,
        rate_code(selected_b_rate) if counts["id_selected"] else 0,
        count_code(counts["switch"]) if counts["id_selected"] else 0,
        2 if counts["back_and_forth"] > 0 else 1 if counts["id_selected"] else 0,
        count_code(counts["id_unmatched"]) if id_fields_complete else 0,
        count_code(counts["id_ambiguous"]) if id_fields_complete else 0,
        rate_code(counts["id_selected"] / max(1, total_frames)) if id_fields_complete else 0,
        rate_code(sum(1 for row in frame_rows if row["selected_discrepancy"] is not None) / max(1, total_frames)) if id_fields_complete else 0,
        count_code(counts["switch_jump_over_threshold"]) if counts["switch"] else 1,
        2 if input_unchanged else 1,
    ]
    return codes, manifest


def output_packet(codes):
    print("CTS001B1-6")
    for index in range(6):
        block = codes[index * 4 : (index + 1) * 4]
        print(f"A{index + 1}-" + "".join(str(value) for value in block))
    print(f"C-{sum(codes) % 10}")


def zero_packet():
    output_packet([0] * 24)


def main():
    path = choose_input_file()
    if path is None or not path.exists():
        zero_packet()
        return
    columns, _ = read_header(path)
    config = configuration_dialog(columns, detect_framecount(columns))
    if not config:
        zero_packet()
        return

    output_dir = Path(__file__).resolve().parent / (
        "camera_target_switch_profile_" + datetime.now().strftime("%Y%m%d_%H%M%S")
    )
    output_dir.mkdir(parents=True, exist_ok=False)
    codes, _ = profile(path, config, output_dir)
    output_packet(codes)


if __name__ == "__main__":
    try:
        main()
    except Exception as exc:
        try:
            Path(__file__).resolve().with_name("camera_target_switch_FATAL.txt").write_text(
                f"{type(exc).__name__}: {exc}\n", encoding="utf-8"
            )
        except Exception:
            pass
        zero_packet()
