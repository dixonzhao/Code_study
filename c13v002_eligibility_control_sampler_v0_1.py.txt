#!/usr/bin/env python3
"""
C13V002 Eligibility and Confident-Control Sampler v0.1

Purpose
-------
Post-process the C13V001 stitched output without changing its selection path.

1. Exclude complete unresolved review windows with a configurable margin.
2. Add an explicit AnalysisEligibility state to every canonical frame.
3. Export a four-column subgraph for human perception.
4. Deterministically select a small set of CONFIDENT, non-window control episodes.
5. Produce an event CSV that can be loaded directly by
   framecount_video_reviewer_v0_2.py.

This program does not select a new Camera target and does not use Radar. It is a
data-quality gate and validation-sampling tool.

Expected input
--------------
- camera13_continuity_stitched_v0_1.csv
- camera13_continuity_review_windows_v0_1.csv

Main outputs
------------
- c13v002_analysis_eligibility_v0_1.csv
- c13v002_confident_control_queue_v0_1.csv
- c13v002_subgraph_v0_1.csv
- C13V002_return_capsule.txt

Review codes in FrameCount Video Reviewer
-----------------------------------------
1 = correct target
2 = wrong target
3 = unsuitable scene / target loss
4 = unclear, timing or source semantics unresolved
5 = video unavailable
"""

from __future__ import annotations

VERSION = "0.1"
NONE_COLUMN = "<NONE>"


@dataclass
class Config:
    frame_column: str
    status_column: str
    source_column: str
    canonical_distance_column: str
    cam1_distance_column: str
    cam3_distance_column: str
    window_id_column: str
    window_start_column: str
    window_end_column: str
    valid_start_frame: float
    maximum_reviewable_frame: float
    exclusion_margin_frames: int
    control_count: int
    control_half_window_frames: int
    minimum_control_separation_frames: int


def normalized_name(value: object) -> str:
    text = "" if value is None else str(value)
    text = re.sub(r"([a-z0-9])([A-Z])", r"\1 \2", text)
    return re.sub(r"[^a-z0-9]+", "", text.lower())


def find_column(
    columns: list[str],
    exact_names: tuple[str, ...] = (),
    required_tokens: tuple[str, ...] = (),
    preferred_tokens: tuple[str, ...] = (),
) -> str:
    normalized = {column: normalized_name(column) for column in columns}
    exact_normalized = {normalized_name(name) for name in exact_names}
    for column, compact in normalized.items():
        if compact in exact_normalized:
            return column

    ranked: list[tuple[int, str]] = []
    for column, compact in normalized.items():
        if required_tokens and not all(normalized_name(token) in compact for token in required_tokens):
            continue
        score = 10 * len(required_tokens)
        score += sum(3 for token in preferred_tokens if normalized_name(token) in compact)
        score -= len(compact) / 10000.0
        ranked.append((score, column))
    return max(ranked, default=(0, NONE_COLUMN))[1]


def choose_file(title: str, patterns: str) -> Path:
    import tkinter as tk
    from tkinter import filedialog

    root = tk.Tk()
    root.withdraw()
    root.update()
    selected = filedialog.askopenfilename(
        title=title,
        filetypes=[("Data files", patterns), ("All files", "*.*")],
    )
    root.destroy()
    if not selected:
        raise RuntimeError(f"No file selected: {title}")
    return Path(selected)


def read_table(path: Path):
    try:
        import pandas as pd
    except ImportError as exc:
        raise RuntimeError("pandas is required in the approved Python environment.") from exc

    if path.suffix.lower() in {".xlsx", ".xls"}:
        return pd.read_excel(path, sheet_name=0, dtype=object)

    encodings = ("utf-8-sig", "utf-8", "cp932", "shift_jis", "latin-1")
    last_error: Exception | None = None
    for encoding in encodings:
        try:
            return pd.read_csv(path, sep=None, engine="python", encoding=encoding)
        except Exception as exc:
            last_error = exc
    raise RuntimeError(f"Could not read {path.name}: {last_error}")

def unique_output_paths(output_dir: Path) -> tuple[Path, Path, Path, Path]:
    names = (
        "c13v002_analysis_eligibility_v0_1.csv",
        "c13v002_confident_control_queue_v0_1.csv",
        "c13v002_subgraph_v0_1.csv",
        "C13V002_return_capsule.txt",
    )
    paths = tuple(output_dir / name for name in names)
    if not any(path.exists() for path in paths):
        return paths  # type: ignore[return-value]
    for number in range(1, 1000):
        suffix = f"_{number:03d}"
        candidate = (
            output_dir / f"c13v002_analysis_eligibility_v0_1{suffix}.csv",
            output_dir / f"c13v002_confident_control_queue_v0_1{suffix}.csv",
            output_dir / f"c13v002_subgraph_v0_1{suffix}.csv",
            output_dir / f"C13V002_return_capsule{suffix}.txt",
        )
        if not any(path.exists() for path in candidate):
            return candidate
    raise RuntimeError("No free output suffix is available.")


def choose_controls(eligible_frame, config: Config):
    import numpy as np
    import pandas as pd

    candidates = eligible_frame[eligible_frame["AnalysisEligibility"] == "ELIGIBLE_PROVISIONAL"].copy()
    frames = pd.to_numeric(candidates[config.frame_column], errors="coerce")
    candidates = candidates[frames.notna()].copy()
    candidates[config.frame_column] = pd.to_numeric(candidates[config.frame_column], errors="coerce")

    lower = config.valid_start_frame + config.control_half_window_frames
    upper = float(candidates[config.frame_column].max()) - config.control_half_window_frames
    if config.maximum_reviewable_frame > 0:
        upper = min(upper, config.maximum_reviewable_frame - config.control_half_window_frames)
    candidates = candidates[
        (candidates[config.frame_column] >= lower)
        & (candidates[config.frame_column] <= upper)
    ].sort_values(config.frame_column, kind="stable")
    if candidates.empty:
        raise RuntimeError("No ELIGIBLE_PROVISIONAL rows are available for control sampling.")

    frame_values = candidates[config.frame_column].to_numpy(dtype=float)
    quantiles = np.linspace(0.1, 0.9, config.control_count)
    target_frames = np.quantile(frame_values, quantiles)
    selected_indices: list[int] = []

    for target in target_frames:
        order = np.argsort(np.abs(frame_values - target), kind="stable")
        chosen: int | None = None
        for candidate_index in order:
            frame_value = frame_values[int(candidate_index)]
            if all(
                abs(frame_value - frame_values[previous])
                >= config.minimum_control_separation_frames
                for previous in selected_indices
            ):
                chosen = int(candidate_index)
                break
        if chosen is not None:
            selected_indices.append(chosen)

    if len(selected_indices) < config.control_count:
        for candidate_index in np.linspace(0, len(candidates) - 1, len(candidates), dtype=int):
            frame_value = frame_values[int(candidate_index)]
            if all(
                abs(frame_value - frame_values[previous])
                >= config.minimum_control_separation_frames
                for previous in selected_indices
            ):
                selected_indices.append(int(candidate_index))
            if len(selected_indices) >= config.control_count:
                break

    selected_indices = sorted(selected_indices[: config.control_count], key=lambda index: frame_values[index])
    rows: list[dict[str, Any]] = []
    for number, candidate_index in enumerate(selected_indices, start=1):
        row = candidates.iloc[candidate_index]
        key_frame = float(row[config.frame_column])
        start_frame = key_frame - config.control_half_window_frames
        end_frame = key_frame + config.control_half_window_frames
        rows.append(
            {
                "EventID": f"Q{number:03d}",
                "StartFrame": int(start_frame) if start_frame.is_integer() else start_frame,
                "KeyFrame": int(key_frame) if key_frame.is_integer() else key_frame,
                "EndFrame": int(end_frame) if end_frame.is_integer() else end_frame,
                "ReasonCode": "CONFIDENT_NONWINDOW_CONTROL",
                "CanonicalCameraSource": row.get(config.source_column, ""),
                "CanonicalCameraDistance": row.get(config.canonical_distance_column, ""),
                "CAM1Distance": row.get(config.cam1_distance_column, ""),
                "CAM3Distance": row.get(config.cam3_distance_column, ""),
                "ReviewResult": "",
                "Memo": "",
            }
        )
    return pd.DataFrame(rows)


def run(
    stitched_path: Path,
    windows_path: Path,
    stitched,
    windows,
    config: Config,
    output_dir: Path,
) -> tuple[Path, Path, Path, Path]:
    import numpy as np
    import pandas as pd

    stitched_required = [
        config.frame_column,
        config.status_column,
        config.source_column,
        config.canonical_distance_column,
        config.cam1_distance_column,
        config.cam3_distance_column,
    ]
    window_required = [
        config.window_id_column,
        config.window_start_column,
        config.window_end_column,
    ]
    missing_stitched = [column for column in stitched_required if column not in stitched.columns]
    missing_windows = [column for column in window_required if column not in windows.columns]
    if missing_stitched:
        raise RuntimeError(f"Missing stitched columns: {missing_stitched}")
    if missing_windows:
        raise RuntimeError(f"Missing review-window columns: {missing_windows}")

    work = stitched.copy()
    work[config.frame_column] = pd.to_numeric(work[config.frame_column], errors="coerce")
    work = work[work[config.frame_column].notna()].copy()
    work = work.sort_values(config.frame_column, kind="stable").reset_index(drop=True)
    frames = work[config.frame_column].to_numpy(dtype=float)

    inside_window = np.zeros(len(work), dtype=bool)
    review_ids = np.full(len(work), "", dtype=object)
    normalized_intervals: list[tuple[str, float, float]] = []
    for _, window in windows.iterrows():
        try:
            start = float(window[config.window_start_column]) - config.exclusion_margin_frames
            end = float(window[config.window_end_column]) + config.exclusion_margin_frames
        except (TypeError, ValueError):
            continue
        if not math.isfinite(start) or not math.isfinite(end):
            continue
        if end < start:
            start, end = end, start
        window_id = str(window.get(config.window_id_column, "")).strip() or "WINDOW"
        left = int(np.searchsorted(frames, start, side="left"))
        right = int(np.searchsorted(frames, end, side="right"))
        if right <= left:
            continue
        inside_window[left:right] = True
        for index in range(left, right):
            if review_ids[index]:
                if window_id not in str(review_ids[index]).split(";"):
                    review_ids[index] = f"{review_ids[index]};{window_id}"
            else:
                review_ids[index] = window_id
        normalized_intervals.append((window_id, start, end))

    statuses = work[config.status_column].astype(str).str.upper().str.strip()
    eligibility: list[str] = []
    reasons: list[str] = []
    for index, (frame_value, status) in enumerate(zip(frames, statuses)):
        if frame_value < config.valid_start_frame:
            eligibility.append("EXCLUDE_PRE_VALID")
            reasons.append("BEFORE_VALID_START")
        elif config.maximum_reviewable_frame > 0 and frame_value > config.maximum_reviewable_frame:
            eligibility.append("EXCLUDE_VIDEO_UNAVAILABLE")
            reasons.append("AFTER_AVAILABLE_VIDEO_END")
        elif status == "INVALID":
            eligibility.append("EXCLUDE_INVALID")
            reasons.append("C13_INVALID")
        elif inside_window[index]:
            eligibility.append("EXCLUDE_REVIEW_WINDOW")
            reasons.append(f"UNRESOLVED_WINDOW:{review_ids[index]}")
        elif status == "CONFIDENT":
            eligibility.append("ELIGIBLE_PROVISIONAL")
            reasons.append("CONFIDENT_OUTSIDE_REVIEW_WINDOWS")
        elif status == "AMBIGUOUS":
            eligibility.append("EXCLUDE_AMBIGUOUS")
            reasons.append("C13_AMBIGUOUS")
        else:
            eligibility.append("EXCLUDE_OTHER")
            reasons.append(f"UNRECOGNIZED_STATUS:{status}")

    work["AnalysisEligibility"] = eligibility
    work["AnalysisEligibilityReason"] = reasons
    work["ReviewWindowIDs"] = review_ids
    work["__C13V002_ExclusionMarginFrames"] = config.exclusion_margin_frames
    work["__C13V002_Method"] = "C13V001_COMPLETE_WINDOW_EXCLUSION_AND_SPACED_CONTROLS"

    controls = choose_controls(work, config)
    subgraph = work[
        [
            config.frame_column,
            config.cam1_distance_column,
            config.cam3_distance_column,
            config.canonical_distance_column,
        ]
    ].copy()
    subgraph.columns = [
        "FrameCount",
        "CAM1Distance",
        "CAM3Distance",
        "CanonicalCameraDistance",
    ]

    output_dir.mkdir(parents=True, exist_ok=True)
    eligibility_path, controls_path, subgraph_path, capsule_path = unique_output_paths(output_dir)
    work.to_csv(eligibility_path, index=False, encoding="utf-8-sig", quoting=csv.QUOTE_MINIMAL)
    controls.to_csv(controls_path, index=False, encoding="utf-8-sig", quoting=csv.QUOTE_MINIMAL)
    subgraph.to_csv(subgraph_path, index=False, encoding="utf-8-sig", quoting=csv.QUOTE_MINIMAL)

    counts = work["AnalysisEligibility"].value_counts().to_dict()
    lines = [
        "C13V002-PREP-v0.1",
        f"STITCHED-{stitched_path.name}",
        f"WINDOWS-{windows_path.name}",
        f"ROWS-{len(work)}",
        f"INPUT_WINDOWS-{len(windows)}",
        f"APPLIED_INTERVALS-{len(normalized_intervals)}",
        f"MARGIN-{config.exclusion_margin_frames}",
        f"VALID_START-{config.valid_start_frame:g}",
        f"VIDEO_END-{config.maximum_reviewable_frame:g}",
        f"ELIGIBLE-{int(counts.get('ELIGIBLE_PROVISIONAL', 0))}",
        f"XWINDOW-{int(counts.get('EXCLUDE_REVIEW_WINDOW', 0))}",
        f"XINVALID-{int(counts.get('EXCLUDE_INVALID', 0))}",
        f"XAMBIGUOUS-{int(counts.get('EXCLUDE_AMBIGUOUS', 0))}",
        f"XVIDEO-{int(counts.get('EXCLUDE_VIDEO_UNAVAILABLE', 0))}",
        f"CONTROLS-{len(controls)}",
        "",
        "CONTROL_KEYS",
    ]
    for _, row in controls.iterrows():
        lines.append(
            f"{row['EventID']}-{row['KeyFrame']}-{row['CanonicalCameraSource']}"
        )
    lines.extend(
        [
            "",
            "NEXT",
            "LOAD-c13v002_confident_control_queue_v0_1.csv-IN-FVR001",
            "RETURN-VIDEO_REVIEW_RETURN_CODE.txt",
            "CODES-1CORRECT-2WRONG-3SCENE-4UNCLEAR-5UNAVAILABLE",
        ]
    )
    capsule_path.write_text("\n".join(lines) + "\n", encoding="utf-8")
    return eligibility_path, controls_path, subgraph_path, capsule_path


def build_cli_config(args, stitched_columns: list[str], window_columns: list[str]) -> Config:
    return Config(
        frame_column=find_column(stitched_columns, ("FrameCount",), ("frame", "count")),
        status_column=find_column(stitched_columns, ("CanonicalSelectionStatus",), ("canonical", "selection", "status")),
        source_column=find_column(stitched_columns, ("CanonicalCameraSource",), ("canonical", "camera", "source")),
        canonical_distance_column=find_column(stitched_columns, ("CanonicalCameraDistance",), ("canonical", "camera", "distance")),
        cam1_distance_column=find_column(stitched_columns, ("ID_221_Distance",), ("distance",), ("221", "cam1", "camera1")),
        cam3_distance_column=find_column(stitched_columns, ("ID_251_Distance",), ("distance",), ("251", "cam3", "camera3")),
        window_id_column=find_column(window_columns, ("WindowID",), ("window", "id")),
        window_start_column=find_column(window_columns, ("StartFrame",), ("start", "frame")),
        window_end_column=find_column(window_columns, ("EndFrame",), ("end", "frame")),
        valid_start_frame=args.valid_start,
        maximum_reviewable_frame=args.video_end,
        exclusion_margin_frames=args.margin,
        control_count=args.controls,
        control_half_window_frames=args.half_window,
        minimum_control_separation_frames=args.minimum_separation,
    )


def main() -> int:
    parser = argparse.ArgumentParser(description="C13V002 eligibility and control sampler")
    parser.add_argument("stitched", nargs="?", help="C13V001 stitched CSV")
    parser.add_argument("windows", nargs="?", help="C13V001 review-window CSV")
    parser.add_argument("--output-dir")
    parser.add_argument("--valid-start", type=float, default=7400)
    parser.add_argument("--video-end", type=float, default=57597)
    parser.add_argument("--margin", type=int, default=20)
    parser.add_argument("--controls", type=int, default=6)
    parser.add_argument("--half-window", type=int, default=40)
    parser.add_argument("--minimum-separation", type=int, default=2000)
    args = parser.parse_args()

    try:
        stitched_path = Path(args.stitched) if args.stitched else choose_file(
            "Select C13V001 stitched output", "*.csv *.txt *.xlsx *.xls"
        )
        windows_path = Path(args.windows) if args.windows else choose_file(
            "Select C13V001 review-window output", "*.csv *.txt *.xlsx *.xls"
        )
        print(f"Reading stitched dataset: {stitched_path}")
        stitched = read_table(stitched_path)
        print(f"Reading review windows: {windows_path}")
        windows = read_table(windows_path)
        if stitched.empty:
            raise RuntimeError("The stitched dataset is empty.")
        if windows.empty:
            raise RuntimeError("The review-window dataset is empty.")

        stitched_columns = [str(column) for column in stitched.columns]
        window_columns = [str(column) for column in windows.columns]
        config = (
            build_cli_config(args, stitched_columns, window_columns)
            if args.stitched and args.windows
            else show_config_dialog(stitched_columns, window_columns)
        )
        required_mappings = [
            config.frame_column,
            config.status_column,
            config.source_column,
            config.canonical_distance_column,
            config.cam1_distance_column,
            config.cam3_distance_column,
            config.window_id_column,
            config.window_start_column,
            config.window_end_column,
        ]
        if any(column == NONE_COLUMN for column in required_mappings):
            raise RuntimeError("One or more required columns could not be detected.")

        output_dir = Path(args.output_dir) if args.output_dir else stitched_path.parent
        outputs = run(stitched_path, windows_path, stitched, windows, config, output_dir)
        print("C13V002 completed.")

            print(output)

        if not (args.stitched and args.windows):
            try:
                import tkinter as tk
                from tkinter import messagebox

                root = tk.Tk()
                root.withdraw()
                messagebox.showinfo(

                    f"After review, return:\nVIDEO_REVIEW_RETURN_CODE.txt",
                )
                root.destroy()
            except Exception:
                pass
        return 0
    except Exception as exc:
        print(f"ERROR: {exc}", file=sys.stderr)
        try:
            import tkinter as tk
            from tkinter import messagebox

            root = tk.Tk()
            root.withdraw()
            messagebox.showerror("C13V002 error", str(exc))
            root.destroy()
        except Exception:
            pass
        return 1
