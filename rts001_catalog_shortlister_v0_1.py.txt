#!/usr/bin/env python3
"""
RTS001 Catalog Shortlister v0.1

Purpose
-------
Reduce a large CAN/DBC signal catalog to a small set of candidates that may
identify the target selected, displayed, or used by an ADAS/ACC subsystem.

This is a read-only discovery utility. It does not modify the input file and
does not decide which signal is correct.

Supported input
---------------
- CSV / TXT (delimiter and common encodings are detected)
- XLSX / XLS (requires pandas and a compatible Excel engine)

Output (beside the selected input file)
----------------------------------------
- RTS001_catalog_candidates_HIGH.csv
- RTS001_catalog_candidates_MEDIUM.csv
- RTS001_catalog_candidates_CONTEXT.csv
- RTS001_catalog_return_capsule.txt
"""

from __future__ import annotations

import csv
import re
import sys
from pathlib import Path
from typing import Iterable


VERSION = "0.1"


def choose_input_file() -> Path:
    try:
        import tkinter as tk
        from tkinter import filedialog

        root = tk.Tk()
        root.withdraw()
        root.update()
        selected = filedialog.askopenfilename(
            title="Select the CAN/DBC signal catalog",
            filetypes=[
                ("Catalog files", "*.csv *.txt *.xlsx *.xls"),
                ("CSV files", "*.csv"),
                ("Excel files", "*.xlsx *.xls"),
                ("All files", "*.*"),
            ],
        )
        root.destroy()
    except Exception as exc:
        raise RuntimeError(f"File-selection dialog could not be opened: {exc}") from exc

    if not selected:
        raise RuntimeError("No input file was selected.")
    return Path(selected)


def read_catalog(path: Path):
    try:
        import pandas as pd
    except ImportError as exc:
        raise RuntimeError(
            "pandas is required. Install it in the approved environment first."
        ) from exc

    suffix = path.suffix.lower()
    if suffix in {".xlsx", ".xls"}:
        frames = []
        sheets = pd.read_excel(path, sheet_name=None, dtype=str)
        for sheet_name, frame in sheets.items():
            frame = frame.copy()
            frame.insert(0, "__SourceSheet", str(sheet_name))
            frames.append(frame)
        if not frames:
            raise RuntimeError("The workbook contains no readable sheets.")
        return pd.concat(frames, ignore_index=True, sort=False).fillna("")

    encodings = ["utf-8-sig", "utf-8", "cp932", "shift_jis", "latin-1"]
    last_error = None
    for encoding in encodings:
        try:
            return pd.read_csv(
                path,
                sep=None,
                engine="python",
                dtype=str,
                encoding=encoding,
                keep_default_na=False,
                on_bad_lines="skip",
            ).fillna("")
        except Exception as exc:
            last_error = exc
    raise RuntimeError(f"The catalog could not be read: {last_error}")


def normalized_text(value: object) -> str:
    text = "" if value is None else str(value)
    text = re.sub(r"([a-z0-9])([A-Z])", r"\1 \2", text)
    text = text.lower()
    text = re.sub(r"[^a-z0-9]+", " ", text)
    return " ".join(text.split())


def has_phrase(text: str, phrase: str) -> bool:
    phrase_normalized = normalized_text(phrase)
    return phrase_normalized in text


def has_token(text: str, token: str) -> bool:
    return re.search(rf"(?:^|\s){re.escape(token)}(?:$|\s)", text) is not None


def score_row(row) -> tuple[int, list[str]]:
    field_texts = [normalized_text(value) for value in row.tolist()]
    text = " | ".join(field_texts)
    reasons: list[str] = []
    score = 0

    # Exact/internal names: strongest evidence.
    compact = re.sub(r"[^a-z0-9]", "", text)
    if "akbpts" in compact:
        score += 120
        reasons.append("AKBPTS_EXACT")
    else:
        akb = has_token(text, "akb")
        pts = has_token(text, "pts")
        if akb:
            score += 35
            reasons.append("AKB")
        if pts:
            score += 30
            reasons.append("PTS")
        if akb and pts:
            score += 40
            reasons.append("AKB_AND_PTS")

    phrase_rules = [
        ("preceding target selection", 70, "PRECEDING_TARGET_SELECTION"),
        ("primary target selection", 70, "PRIMARY_TARGET_SELECTION"),
        ("selected target", 55, "SELECTED_TARGET"),
        ("target selection", 50, "TARGET_SELECTION"),
        ("control target", 50, "CONTROL_TARGET"),
        ("display target", 45, "DISPLAY_TARGET"),
        ("reference target", 45, "REFERENCE_TARGET"),
        ("target channel", 42, "TARGET_CHANNEL"),
        ("camera channel", 38, "CAMERA_CHANNEL"),
        ("target id", 42, "TARGET_ID"),
        ("object id", 38, "OBJECT_ID"),
        ("track id", 38, "TRACK_ID"),
        ("lead vehicle", 35, "LEAD_VEHICLE"),
        ("preceding vehicle", 35, "PRECEDING_VEHICLE"),
        ("vehicle symbol", 28, "VEHICLE_SYMBOL"),
        ("vehicle detect", 25, "VEHICLE_DETECT"),
    ]
    for phrase, points, reason in phrase_rules:
        if has_phrase(text, phrase):
            score += points
            reasons.append(reason)

    token_rules = [
        ("target", 12, "TARGET"),
        ("tgt", 12, "TGT"),
        ("selected", 14, "SELECTED"),
        ("select", 14, "SELECT"),
        ("selector", 14, "SELECTOR"),
        ("sel", 8, "SEL"),
        ("control", 8, "CONTROL"),
        ("ctrl", 8, "CTRL"),
        ("reference", 7, "REFERENCE"),
        ("recognition", 8, "RECOGNITION"),
        ("display", 7, "DISPLAY"),
        ("preceding", 10, "PRECEDING"),
        ("primary", 8, "PRIMARY"),
        ("lead", 8, "LEAD"),
        ("following", 6, "FOLLOWING"),
        ("object", 5, "OBJECT"),
        ("track", 6, "TRACK"),
        ("camera", 4, "CAMERA"),
        ("acc", 5, "ACC"),
        ("id", 4, "ID"),
        ("channel", 4, "CHANNEL"),
        ("ch", 2, "CH"),
        ("switch", 5, "SWITCH"),
        ("switching", 5, "SWITCHING"),
    ]
    for token, points, reason in token_rules:
        if has_token(text, token):
            score += points
            reasons.append(reason)

    # ACC/HMI message neighborhood is context, not proof. It receives only a
    # modest boost and still needs target/selection semantics to rank highly.
    if re.search(r"(?:0x)?5e[0-4]\b", compact):
        score += 5
        reasons.append("ACC_HMI_ID_NEIGHBORHOOD")

    # Compound evidence is more informative than isolated generic words.
    has_target_semantic = any(
        reason in reasons
        for reason in ("TARGET", "TGT", "OBJECT", "PRECEDING", "LEAD")
    )
    has_selection_semantic = any(
        reason in reasons
        for reason in ("SELECTED", "SELECT", "SELECTOR", "SEL", "SWITCH", "SWITCHING")
    )
    has_identity_semantic = any(
        reason in reasons for reason in ("ID", "CHANNEL", "CH", "RECOGNITION")
    )
    if has_target_semantic and has_selection_semantic:
        score += 25
        reasons.append("TARGET_PLUS_SELECTION")
    if has_target_semantic and has_identity_semantic:
        score += 15
        reasons.append("TARGET_PLUS_IDENTITY")

    return score, reasons


def write_csv(frame, path: Path) -> None:
    frame.to_csv(path, index=False, encoding="utf-8-sig", quoting=csv.QUOTE_MINIMAL)


def compact_value(value: object, limit: int = 55) -> str:
    text = " ".join(str(value).replace("\r", " ").replace("\n", " ").split())
    return text if len(text) <= limit else text[: limit - 3] + "..."


def choose_summary_columns(columns: Iterable[object]) -> list[str]:
    names = [str(column) for column in columns]
    preferred_patterns = [
        r"signal",
        r"message",
        r"can.*id|^id$",
        r"channel|^ch$",
        r"comment|description|remark",
        r"enum|value.*table|choice",
        r"unit",
        r"minimum|maximum|^min$|^max$",
    ]
    selected: list[str] = []
    for pattern in preferred_patterns:
        for name in names:
            if name in selected or name.startswith("__RTS001_"):
                continue
            if re.search(pattern, normalized_text(name)):
                selected.append(name)
                break
    if not selected:
        selected = [name for name in names if not name.startswith("__RTS001_")][:6]
    return selected[:8]


def main() -> int:
    try:
        input_path = choose_input_file()
        frame = read_catalog(input_path)
        if frame.empty:
            raise RuntimeError("The selected catalog contains no rows.")

        scored = frame.copy()
        scores_and_reasons = scored.apply(score_row, axis=1)
        scored.insert(0, "__RTS001_Score", [item[0] for item in scores_and_reasons])
        scored.insert(1, "__RTS001_Reasons", [";".join(item[1]) for item in scores_and_reasons])
        scored.insert(2, "__RTS001_OriginalRow", [index + 2 for index in range(len(scored))])
        scored = scored.sort_values(
            by=["__RTS001_Score", "__RTS001_OriginalRow"],
            ascending=[False, True],
            kind="stable",
        )

        high = scored[scored["__RTS001_Score"] >= 45].copy()
        medium = scored[
            (scored["__RTS001_Score"] >= 20)
            & (scored["__RTS001_Score"] < 45)
        ].copy()
        context = scored[
            (scored["__RTS001_Score"] > 0)
            & (scored["__RTS001_Score"] < 20)
        ].copy()

        output_dir = input_path.parent
        high_path = output_dir / "RTS001_catalog_candidates_HIGH.csv"
        medium_path = output_dir / "RTS001_catalog_candidates_MEDIUM.csv"
        context_path = output_dir / "RTS001_catalog_candidates_CONTEXT.csv"
        capsule_path = output_dir / "RTS001_catalog_return_capsule.txt"
        write_csv(high, high_path)
        write_csv(medium, medium_path)
        write_csv(context, context_path)

        summary_columns = choose_summary_columns(scored.columns)
        lines = [
            f"RTS001-CATALOG-SHORTLIST-v{VERSION}",
            f"INPUT_ROWS-{len(frame)}",
            f"HIGH-{len(high)}",
            f"MEDIUM-{len(medium)}",
            f"CONTEXT-{len(context)}",
            "",
            "TOP-CANDIDATES",
        ]
        for rank, (_, row) in enumerate(scored.head(24).iterrows(), start=1):
            parts = [
                f"R{rank:02d}",
                f"S{int(row['__RTS001_Score'])}",
                compact_value(row["__RTS001_Reasons"], 80),
            ]
            for column in summary_columns:
                value = compact_value(row.get(column, ""))
                if value:
                    parts.append(f"{compact_value(column, 20)}={value}")
            lines.append("|".join(parts))

        lines.extend(
            [
                "",
                "INTERPRETATION",
                "HIGH rows are lexical candidates, not confirmed target-selector signals.",
                "Return the complete capsule text and manually inspect HIGH before MEDIUM.",
            ]
        )
        capsule_path.write_text("\n".join(lines) + "\n", encoding="utf-8")

        print("RTS001 catalog shortlisting completed.")
        print(f"Input rows: {len(frame)}")
        print(f"HIGH: {len(high)}")
        print(f"MEDIUM: {len(medium)}")
        print(f"CONTEXT: {len(context)}")
        print(f"Return capsule: {capsule_path}")

        try:
            import tkinter as tk
            from tkinter import messagebox

            root = tk.Tk()
            root.withdraw()
            messagebox.showinfo(
                "RTS001 completed",
                "Catalog shortlisting completed.\n\n"
                f"HIGH: {len(high)}\n"
                f"MEDIUM: {len(medium)}\n"
                f"CONTEXT: {len(context)}\n\n"
                "Please return RTS001_catalog_return_capsule.txt.",
            )
            root.destroy()
        except Exception:
            pass
        return 0
    except Exception as exc:
        print(f"ERROR: {exc}", file=sys.stderr)
        try:
            import tkinter as tk
            from tkinter import messagebox

            root = tk.Tk()
            root.withdraw()
            messagebox.showerror("RTS001 error", str(exc))
            root.destroy()
        except Exception:
            pass
        return 1


if __name__ == "__main__":
    raise SystemExit(main())
