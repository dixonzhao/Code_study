from __future__ import annotations

import argparse
import csv
from collections import Counter
from pathlib import Path


ALLOWED_KINDS = ("FLOAT", "INTEGER", "BOOLEAN", "ENUM", "BITFIELD")
REQUIRED_COLUMNS = ("CanonicalName", "ValueKind", "SignalRole")


def audit(path: Path, expected_total: int | None, expected_frame_rows: int | None) -> int:
    counts: Counter[str] = Counter()
    frame_names: set[str] = set()
    total = 0
    missing_semantics = 0
    frame_rows = 0
    frame_integer_rows = 0
    frame_kind_mismatch = 0

    with path.open("r", encoding="utf-8-sig", newline="") as handle:
        reader = csv.DictReader(handle)
        if not reader.fieldnames:
            print("UDP001G4A1")
            print("C-2")
            print("M1-empty-or-missing-header")
            return 2

        missing_columns = [name for name in REQUIRED_COLUMNS if name not in reader.fieldnames]
        if missing_columns:
            print("UDP001G4A1")
            print("C-2")
            print("M1-missing-columns-" + "_".join(missing_columns))
            return 2

        for row in reader:
            total += 1
            kind = (row.get("ValueKind") or "").strip().upper()
            role = (row.get("SignalRole") or "").strip().upper()
            canonical_name = (row.get("CanonicalName") or "").strip()

            if kind in ALLOWED_KINDS:
                counts[kind] += 1
            else:
                counts["OTHER"] += 1

            if not kind or not role or not canonical_name:
                missing_semantics += 1

            if role == "FRAME_KEY":
                frame_rows += 1
                frame_names.add(canonical_name)
                if kind == "INTEGER":
                    frame_integer_rows += 1
                else:
                    frame_kind_mismatch += 1

    allowed_sum = sum(counts[name] for name in ALLOWED_KINDS)
    total_matches = expected_total is None or total == expected_total
    frame_matches = expected_frame_rows is None or frame_rows == expected_frame_rows

    passed = all(
        (
            total > 0,
            allowed_sum == total,
            counts["OTHER"] == 0,
            missing_semantics == 0,
            len(frame_names) == 1,
            frame_rows > 0,
            frame_integer_rows == frame_rows,
            frame_kind_mismatch == 0,
            total_matches,
            frame_matches,
        )
    )

    print("UDP001G4A1")
    print(
        "N-"
        f"U{total}-S{allowed_sum}-"
        f"F{counts['FLOAT']}-I{counts['INTEGER']}-B{counts['BOOLEAN']}-"
        f"E{counts['ENUM']}-H{counts['BITFIELD']}-O{counts['OTHER']}"
    )
    print(
        "R-"
        f"K{len(frame_names)}-T{frame_rows}-J{frame_integer_rows}-"
        f"W{frame_kind_mismatch}-M{missing_semantics}"
    )
    print(f"C-{0 if passed else 1}")
    return 0 if passed else 1


def main() -> int:
    parser = argparse.ArgumentParser(
        description="Read-only Gate 4 semantic count conservation audit."
    )
    parser.add_argument("--updates", required=True, type=Path)
    parser.add_argument("--expected-total", type=int)
    parser.add_argument("--expected-frame-rows", type=int)
    args = parser.parse_args()

    if not args.updates.is_file():
        print("UDP001G4A1")
        print("C-2")
        print("M1-input-file-not-found")
        return 2

    return audit(args.updates, args.expected_total, args.expected_frame_rows)


if __name__ == "__main__":
    raise SystemExit(main())
