def normalized(value) -> str:
    return re.sub(r"[^a-z0-9]", "", str(value).lower())

def sniff_csv(path: Path):
    raw = path.read_bytes()[:131072]
    decoded = None
    encoding = None
    for candidate in ["utf-8-sig", "cp932", "shift_jis", "utf-8", "latin1"]:
        try:
            decoded = raw.decode(candidate)
            encoding = candidate
            break
        except UnicodeDecodeError:
            continue
    if decoded is None:
        decoded = raw.decode("latin1", errors="replace")
        encoding = "latin1"
    try:
        separator = csv.Sniffer().sniff(decoded, delimiters=",;\t|").delimiter
    except Exception:
        counts = {item: decoded.count(item) for item in [",", ";", "\t", "|"]}
        separator = max(counts, key=counts.get)
    return encoding, separator


def read_header(path: Path):
    import pandas as pd

    if path.suffix.lower() in {".xlsx", ".xls", ".xlsm"}:
        frame = pd.read_excel(path, nrows=0)
        return list(frame.columns), {"kind": "excel"}
    encoding, separator = sniff_csv(path)
    frame = pd.read_csv(path, sep=separator, encoding=encoding, nrows=0)
    return list(frame.columns), {
        "kind": "csv",
        "encoding": encoding,
        "separator": separator,
    }


def parse_sentinels(text: str):
    values = []
    for item in re.split(r"[,;\s]+", text.strip()):
        if not item:
            continue
        try:
            values.append(float(item))
        except ValueError:
            pass
    return sorted(set(values))


def has_can_id(column, can_id):
    raw = str(column).lower()
    hex_id = f"{can_id:x}"
    patterns = [
        rf"0x0*{re.escape(hex_id)}(?![0-9a-f])",
        rf"\bid0*{re.escape(hex_id)}(?![0-9a-f])",
        rf"(?<![0-9a-f])0*{re.escape(hex_id)}(?![0-9a-f])",
    ]
    return any(re.search(pattern, raw) for pattern in patterns)


def role_match_score(column, role):
    key = normalized(column)
    score = 0
    tokens = ROLE_TOKENS[role]
    if any(token in key for token in tokens):
        score += 50
    if role == "distance" and any(
        token in key for token in ["lateral", "relative", "width"]
    ):
        score -= 80
    if role == "relative_speed" and "lateral" in key:
        score -= 80
    return score


def guess_camera_column(columns, can_id, role):
    scored = []
    for column in columns:
        score = role_match_score(column, role)
        if has_can_id(column, can_id):
            score += 100
        else:
            score -= 100
        if score > 100:
            scored.append((score, str(column)))
    scored.sort(reverse=True)
    if not scored:
        return None
    if len(scored) > 1 and scored[0][0] == scored[1][0]:
        return None
    return scored[0][1]


def guess_global_column(columns, can_id, include_tokens):
    scored = []
    for column in columns:
        key = normalized(column)
        score = 100 if has_can_id(column, can_id) else 0
        if all(token in key for token in include_tokens):
            score += 60
        if score >= 160:
            scored.append((score, str(column)))
    scored.sort(reverse=True)
    if len(scored) == 1 or (
        len(scored) > 1 and scored[0][0] > scored[1][0]
    ):
        return scored[0][1]
    return None


def guess_frame_column(columns):
    candidates = [column for column in columns if "framecount" in normalized(column)]
    return str(candidates[0]) if len(candidates) == 1 else None

def fast_fingerprint(path: Path):
    digest = hashlib.sha256()
    size = path.stat().st_size
    digest.update(str(size).encode("ascii"))
    with path.open("rb") as handle:
        digest.update(handle.read(1024 * 1024))
        if size > 1024 * 1024:
            handle.seek(max(0, size - 1024 * 1024))
            digest.update(handle.read(1024 * 1024))
    return digest.hexdigest()

def valid_number_mask(series, sentinels):
    import numpy as np

    values = series.to_numpy(dtype=float, na_value=np.nan)
    mask = np.isfinite(values)
    for sentinel in sentinels:
        mask &= ~np.isclose(values, sentinel, rtol=1e-12, atol=1e-12, equal_nan=False)
    return mask


def near_one_frame(mask):
    return mask | mask.shift(1, fill_value=False) | mask.shift(-1, fill_value=False)


def rate_code(rate):
    if rate <= 0:
        return 1
    if rate < 0.10:
        return 2
    if rate < 0.90:
        return 3
    return 4


def count_code(count):
    if count == 0:
        return 1
    if count <= 10:
        return 2
    if count <= 100:
        return 3
    return 4


def boolean_rate(mask, denominator_mask=None):
    if denominator_mask is None:
        return float(mask.mean()) if len(mask) else 0.0
    denominator = int(denominator_mask.sum())
    return float((mask & denominator_mask).sum() / denominator) if denominator else 0.0


def safe_number(value):
    if value is None:
        return None
    try:
        value = float(value)
    except (TypeError, ValueError):
        return None
    return value if math.isfinite(value) else None


def top_values(series, limit=20):
    values = []
    for value, count in series.dropna().value_counts().head(limit).items():
        values.append({"value": safe_number(value), "frame_count": int(count)})
    return values


def flattened_mapping(config):
    roles = {}
    for role, column in config["global_mapping"].items():
        if column:
            roles[f"global__{role}"] = column
    for can_id, mapping in config["camera_mapping"].items():
        for role, column in mapping.items():
            if column:
                roles[f"cam_{int(can_id):03x}__{role}"] = column
    return roles


def build_selector(path: Path, config: dict, output_dir: Path):
    import numpy as np
    import pandas as pd

    output_dir.mkdir(parents=True, exist_ok=True)
    _, reader_info = read_header(path)
    role_to_column = flattened_mapping(config)
    selected_columns = list(dict.fromkeys(role_to_column.values()))
    before_stat = path.stat()
    started = time.perf_counter()

    source = read_selected(path, selected_columns, reader_info)
    total_input_rows = len(source)
    source = source.rename(columns={column: role for role, column in role_to_column.items()})
    for role in role_to_column:
        source[role] = pd.to_numeric(source[role], errors="coerce")

    frame_role = "global__frame"
    source = source[source[frame_role].notna()].copy()
    source[frame_role] = source[frame_role].astype(float)
    raw_frames = source[frame_role].to_numpy()
    backward_transitions = int(np.sum(np.diff(raw_frames) < 0)) if len(raw_frames) > 1 else 0

    frame_end = source.drop_duplicates(frame_role, keep="last").set_index(frame_role, drop=False)
    frame_end.index.name = "_frame_index"
    frame_index = frame_end.index

    global_conflict_roles = [
        role
        for role in [
            "global__recognition_id",
            "global__radar_distance",
            "global__radar_detection",
        ]
        if role in source.columns
    ]
    global_conflicts = (
        source.groupby(frame_role, sort=False)[global_conflict_roles]
        .nunique(dropna=True)
        .gt(1)
        .reindex(frame_index, fill_value=False)
    )

    candidate_states = {}
    candidate_diagnostics = {}
    multi_update_frames = 0
    for can_id in CAMERA_IDS:
        prefix = f"cam_{can_id:03x}__"
        counter_role = prefix + "message_counter"
        counter = source[counter_role]
        previous_counter = counter.shift(1)
        update_mask = counter.notna() & (
            previous_counter.isna() | counter.ne(previous_counter)
        )
        mapped_roles = [
            prefix + role
            for role in CAMERA_ROLES
            if prefix + role in source.columns
        ]
        event_columns = [frame_role] + mapped_roles
        events = source.loc[update_mask, event_columns].copy()
        update_counts = events.groupby(frame_role, sort=False).size()
        multi_update_frames += int((update_counts > 1).sum())
        last_event = events.drop_duplicates(frame_role, keep="last").set_index(frame_role)
        state = last_event.reindex(frame_index).ffill()

        last_update_frame = pd.Series(np.nan, index=frame_index, dtype=float)
        if not last_event.empty:
            last_update_frame.loc[last_event.index] = last_event.index.astype(float)
        last_update_frame = last_update_frame.ffill()
        age = pd.Series(frame_index.to_numpy(dtype=float), index=frame_index) - last_update_frame
        updated_this_frame = pd.Series(frame_index.isin(last_event.index), index=frame_index)
        updates_in_frame = update_counts.reindex(frame_index, fill_value=0).astype(int)

        candidate_states[can_id] = state
        candidate_diagnostics[can_id] = {
            "last_update_frame": last_update_frame,
            "signal_age_frames": age,
            "updated_this_frame": updated_this_frame,
            "updates_in_frame": updates_in_frame,
            "total_update_events": int(len(events)),
        }

    result = pd.DataFrame(index=frame_index)
    result["FrameCount"] = frame_index.to_numpy(dtype=float)
    for role in [
        "radar_distance",
        "radar_relative_speed",
        "radar_detection",
        "recognition_id",
        "vehicle_detect",
        "acc_system_status",
        "halt_state",
        "auto_adjust_complete",
    ]:
        source_role = f"global__{role}"
        if source_role in frame_end.columns:
            result[role] = frame_end[source_role]
        elif role in {
            "radar_relative_speed",
            "vehicle_detect",
            "acc_system_status",
            "halt_state",
            "auto_adjust_complete",
        }:
            result[role] = np.nan

    for role in global_conflict_roles:
        short = role.replace("global__", "")
        result[f"{short}_intra_frame_conflict"] = global_conflicts[role].astype(bool)

    result["region"] = np.select(
        [
            result["FrameCount"] < config["valid_start"],
            result["FrameCount"] < config["switch_start"],
        ],
        ["PRE_VALID", "STABLE"],
        default="INVESTIGATION",
    )

    recognition = result["recognition_id"]
    recognition_int = recognition.round().astype("Int64")
    result["SelectedCameraCANID_Decimal"] = recognition_int
    result["SelectedCameraCANID_Hex"] = recognition_int.map(
        lambda value: f"0x{int(value):03X}" if pd.notna(value) else ""
    )

    selected_columns = {
        "distance": "SelectedCameraDistance",
        "relative_speed": "SelectedCameraRelativeSpeed",
        "lateral_position": "SelectedCameraLateralPosition",
        "switching_flag": "SelectedCameraSwitchingFlag",
        "object_detection_type": "SelectedCameraObjectDetectionType",
        "message_counter": "SelectedCameraMessageCounter",
    }
    for output_column in selected_columns.values():
        result[output_column] = np.nan
    result["SelectedCameraLastUpdateFrame"] = np.nan
    result["SelectedCameraSignalAgeFrames"] = np.nan
    result["SelectedCameraUpdatedThisFrame"] = False
    result["SelectedCameraUpdatesInFrame"] = 0

    selected_can_masks = {}
    for can_id in CAMERA_IDS:
        mask = recognition_int.eq(can_id).fillna(False)
        selected_can_masks[can_id] = mask
        state = candidate_states[can_id]
        prefix = f"cam_{can_id:03x}__"
        for role, output_column in selected_columns.items():
            source_role = prefix + role
            if source_role in state.columns:
                result.loc[mask, output_column] = state.loc[mask, source_role]
        diagnostics = candidate_diagnostics[can_id]
        result.loc[mask, "SelectedCameraLastUpdateFrame"] = diagnostics[
            "last_update_frame"
        ].loc[mask]
        result.loc[mask, "SelectedCameraSignalAgeFrames"] = diagnostics[
            "signal_age_frames"
        ].loc[mask]
        result.loc[mask, "SelectedCameraUpdatedThisFrame"] = diagnostics[
            "updated_this_frame"
        ].loc[mask]
        result.loc[mask, "SelectedCameraUpdatesInFrame"] = diagnostics[
            "updates_in_frame"
        ].loc[mask]
        result[f"Camera0x{can_id:03X}_UpdatedThisFrame"] = diagnostics[
            "updated_this_frame"
        ].astype(bool)
        result[f"Camera0x{can_id:03X}_SignalAgeFrames"] = diagnostics[
            "signal_age_frames"
        ]

    supported_mask = pd.Series(False, index=result.index)
    for mask in selected_can_masks.values():
        supported_mask |= mask
    missing_recognition = recognition.isna()
    camera_distance_valid = pd.Series(
        valid_number_mask(
            result["SelectedCameraDistance"], config["camera_sentinels"]
        ),
        index=result.index,
    )
    has_candidate_state = result["SelectedCameraLastUpdateFrame"].notna()
    stale_mask = pd.Series(False, index=result.index)
    if config["max_signal_age_frames"] is not None:
        stale_mask = result["SelectedCameraSignalAgeFrames"].gt(
            config["max_signal_age_frames"]
        )

    result["CameraSelectionStatus"] = "SELECTED_VALID"
    result.loc[~supported_mask & ~missing_recognition, "CameraSelectionStatus"] = (
        "UNSUPPORTED_RECOGNITION_ID"
    )
    result.loc[missing_recognition, "CameraSelectionStatus"] = "MISSING_RECOGNITION_ID"
    result.loc[
        supported_mask & ~has_candidate_state, "CameraSelectionStatus"
    ] = "NO_CANDIDATE_UPDATE_OBSERVED"
    result.loc[
        supported_mask & has_candidate_state & ~camera_distance_valid,
        "CameraSelectionStatus",
    ] = "INVALID_SELECTED_DISTANCE"
    result.loc[
        supported_mask & has_candidate_state & camera_distance_valid & stale_mask,
        "CameraSelectionStatus",
    ] = "STALE_SELECTED_SIGNAL"
    result.loc[result["region"].eq("PRE_VALID"), "CameraSelectionStatus"] = (
        "PRE_VALID_FRAME"
    )

    selected_valid = result["CameraSelectionStatus"].eq("SELECTED_VALID")
    radar_distance_valid = pd.Series(
        valid_number_mask(result["radar_distance"], config["radar_sentinels"]),
        index=result.index,
    )
    radar_detected = np.isclose(
        result["radar_detection"],
        config["radar_detected_value"],
        rtol=0,
        atol=1e-12,
        equal_nan=False,
    )
    result["RadarDetected"] = radar_detected
    discrepancy_gate = selected_valid & radar_detected & radar_distance_valid
    result["ProvisionalDiscrepancy"] = np.where(
        discrepancy_gate,
        result["radar_distance"] - result["SelectedCameraDistance"],
        np.nan,
    )

    previous_supported_id = recognition_int.where(supported_mask).shift(1)
    transition_mask = (
        recognition_int.notna()
        & previous_supported_id.notna()
        & recognition_int.ne(previous_supported_id)
        & supported_mask
    )
    result["RecognitionTransition"] = transition_mask
    result["PreviousSupportedCameraCANID"] = previous_supported_id
    any_camera_update = pd.Series(False, index=result.index)
    for diagnostics in candidate_diagnostics.values():
        any_camera_update |= diagnostics["updated_this_frame"]
    result["AnyCameraUpdateNear"] = near_one_frame(any_camera_update)

    transition_columns = [
        "FrameCount",
        "region",
        "PreviousSupportedCameraCANID",
        "SelectedCameraCANID_Decimal",
        "SelectedCameraCANID_Hex",
        "recognition_id_intra_frame_conflict",
        "SelectedCameraUpdatedThisFrame",
        "SelectedCameraUpdatesInFrame",
        "SelectedCameraSignalAgeFrames",
        "AnyCameraUpdateNear",
        "SelectedCameraDistance",
        "SelectedCameraRelativeSpeed",
        "SelectedCameraLateralPosition",
        "SelectedCameraSwitchingFlag",
        "SelectedCameraObjectDetectionType",
        "radar_distance",
        "radar_relative_speed",
        "RadarDetected",
        "CameraSelectionStatus",
        "ProvisionalDiscrepancy",
    ]
    transitions = result.loc[transition_mask, transition_columns].copy()
    unsupported = result.loc[
        result["CameraSelectionStatus"].eq("UNSUPPORTED_RECOGNITION_ID"),
        [
            "FrameCount",
            "region",
            "recognition_id",
            "SelectedCameraCANID_Decimal",
            "SelectedCameraCANID_Hex",
            "recognition_id_intra_frame_conflict",
        ],
    ].copy()

    pre_mask = result["region"].eq("PRE_VALID")
    stable_mask = result["region"].eq("STABLE")
    investigation_mask = result["region"].eq("INVESTIGATION")
    valid_region = ~pre_mask
    status_counts = Counter(result.loc[valid_region, "CameraSelectionStatus"])
    selection_counts = {
        f"0x{can_id:03X}": int((recognition_int.eq(can_id) & valid_region).sum())
        for can_id in CAMERA_IDS
    }
    selected_age = result.loc[selected_valid & valid_region, "SelectedCameraSignalAgeFrames"]
    age_q95 = safe_number(selected_age.quantile(0.95))

    metrics = {
        "total_input_rows": int(total_input_rows),
        "total_frame_records": int(len(result)),
        "backward_frame_transitions": backward_transitions,
        "pre_valid_frame_records": int(pre_mask.sum()),
        "stable_frame_records": int(stable_mask.sum()),
        "investigation_frame_records": int(investigation_mask.sum()),
        "recognition_nonmissing_rate_valid": boolean_rate(
            recognition.notna(), valid_region
        ),
        "recognition_supported_rate_valid": boolean_rate(
            supported_mask, valid_region
        ),
        "recognition_unsupported_rate_valid": boolean_rate(
            ~supported_mask & recognition.notna(), valid_region
        ),
        "selector_valid_rate_valid": boolean_rate(selected_valid, valid_region),
        "selector_valid_rate_stable": boolean_rate(selected_valid, stable_mask),
        "selector_valid_rate_investigation": boolean_rate(
            selected_valid, investigation_mask
        ),
        "radar_detected_rate_valid": boolean_rate(
            pd.Series(radar_detected, index=result.index), valid_region
        ),
        "provisional_discrepancy_rate_valid": boolean_rate(
            result["ProvisionalDiscrepancy"].notna(), valid_region
        ),
        "recognition_transition_count": int(transition_mask.sum()),
        "transition_any_camera_update_near_rate": boolean_rate(
            result["AnyCameraUpdateNear"], transition_mask
        ),
        "selected_updated_this_frame_rate": boolean_rate(
            result["SelectedCameraUpdatedThisFrame"], selected_valid & valid_region
        ),
        "recognition_conflict_frame_count": int(
            result["recognition_id_intra_frame_conflict"].sum()
        ),
        "multi_update_candidate_frame_count": int(multi_update_frames),
        "selected_age_nonmissing_rate": boolean_rate(
            result["SelectedCameraSignalAgeFrames"].notna(), selected_valid & valid_region
        ),
        "selected_signal_age_median": safe_number(selected_age.median()),
        "selected_signal_age_q95": age_q95,
        "selected_signal_age_max": safe_number(selected_age.max()),
        "provisional_discrepancy_median": safe_number(
            result.loc[valid_region, "ProvisionalDiscrepancy"].median()
        ),
        "provisional_discrepancy_q95_abs": safe_number(
            result.loc[valid_region, "ProvisionalDiscrepancy"].abs().quantile(0.95)
        ),
    }

    after_stat = path.stat()
    input_unchanged = (
        after_stat.st_size == before_stat.st_size
        and after_stat.st_mtime_ns == before_stat.st_mtime_ns
    )

    canonical_path = output_dir / "multi_camera_canonical_frame_v0_1.csv"
    transition_path = output_dir / "multi_camera_reference_transitions_v0_1.csv"
    unsupported_path = output_dir / "multi_camera_unsupported_references_v0_1.csv"
    manifest_path = output_dir / "multi_camera_selector_manifest_v0_1.json"
    log_path = output_dir / "multi_camera_selector_log_v0_1.txt"

    result.reset_index(drop=True).to_csv(canonical_path, index=False, encoding="utf-8-sig")
    transitions.to_csv(transition_path, index=False, encoding="utf-8-sig")
    unsupported.to_csv(unsupported_path, index=False, encoding="utf-8-sig")

    manifest = {
        "version": VERSION,
        "run_id": str(uuid.uuid4()),
        "created_utc": datetime.now(timezone.utc).isoformat(),
        "input": {
            "basename": path.name,
            "size_bytes": before_stat.st_size,
            "fast_fingerprint_sha256": fast_fingerprint(path),
            "input_unchanged": input_unchanged,
        },
        "configuration": {
            key: value for key, value in config.items() if key != "confirmed"
        },
        "registry": {
            f"0x{can_id:03X}": config["camera_mapping"][str(can_id)]
            for can_id in CAMERA_IDS
        },
        "metrics": metrics,
        "selection_counts": selection_counts,
        "selection_status_counts_valid_region": dict(status_counts),
        "recognition_id_top20": top_values(recognition),
        "candidate_update_event_counts": {
            f"0x{can_id:03X}": diagnostics["total_update_events"]
            for can_id, diagnostics in candidate_diagnostics.items()
        },
        "interpretation_limits": [
            "Recognition_ID decimal CAN-ID mapping remains an engineering selector hypothesis until business/DBC confirmation.",
            "The selector never substitutes another Camera candidate when Recognition_ID is unsupported, missing, invalid, or stale.",
            "Radar is a provisional reference and has no target ID in the current dataset.",
            "ProvisionalDiscrepancy is not Camera ground-truth error.",
            "MessageCounter changes reconstruct Camera update provenance after P002 forward filling.",
            "The first observed counter value is treated as an initial update event; age near the file start requires caution.",
            "ObjectDetectionType and SwitchingFlag are exported but are not validity gates until their semantics are confirmed.",
            "FrameCount 38239 is an investigation label, not a selector rule.",
        ],
    }
    manifest_path.write_text(
        json.dumps(manifest, ensure_ascii=False, indent=2), encoding="utf-8"
    )
    log_path.write_text(
        "\n".join(
            [
                f"version={VERSION}",
                f"input_basename={path.name}",
                f"elapsed_seconds={time.perf_counter() - started:.3f}",
                f"input_unchanged={input_unchanged}",
                f"metrics={metrics}",
                f"selection_counts={selection_counts}",
                f"selection_status_counts={dict(status_counts)}",
            ]
        )
        + "\n",
        encoding="utf-8",
    )

    total_frames = len(result)
    frame_bucket = 1 if total_frames < 50_000 else 2 if total_frames < 200_000 else 3
    age_code = (
        1
        if age_q95 is None or age_q95 > 20
        else 2
        if age_q95 > 5
        else 3
        if age_q95 > 1
        else 4
    )
    stale_rate = (
        boolean_rate(
            result["CameraSelectionStatus"].eq("STALE_SELECTED_SIGNAL"), valid_region
        )
        if config["max_signal_age_frames"] is not None
        else None
    )
    optional_total = len(CAMERA_IDS) * len(CAMERA_OPTIONAL_ROLES) + 5
    optional_selected = sum(
        1
        for can_id in CAMERA_IDS
        for role in CAMERA_OPTIONAL_ROLES
        if config["camera_mapping"][str(can_id)].get(role)
    ) + sum(
        1
        for role in [
            "radar_relative_speed",
            "vehicle_detect",
            "acc_system_status",
            "halt_state",
            "auto_adjust_complete",
        ]
        if config["global_mapping"].get(role)
    )
    appeared_codes = [
        2 if selection_counts[f"0x{can_id:03X}"] > 0 else 1 for can_id in CAMERA_IDS
    ]
    codes = [
        2,
        2,
        2,
        2 if input_unchanged else 1,
        2 if backward_transitions == 0 else 1,
        frame_bucket,
        2 if pre_mask.any() else 1,
        2 if investigation_mask.any() else 1,
        rate_code(metrics["recognition_nonmissing_rate_valid"]),
        rate_code(metrics["recognition_supported_rate_valid"]),
        rate_code(metrics["selector_valid_rate_valid"]),
        rate_code(metrics["recognition_unsupported_rate_valid"]),
        rate_code(metrics["selector_valid_rate_stable"]),
        rate_code(metrics["selector_valid_rate_investigation"]),
        rate_code(metrics["radar_detected_rate_valid"]),
        rate_code(metrics["provisional_discrepancy_rate_valid"]),
        count_code(metrics["recognition_transition_count"]),
        rate_code(metrics["transition_any_camera_update_near_rate"]),
        rate_code(metrics["selected_updated_this_frame_rate"]),
        count_code(metrics["recognition_conflict_frame_count"]),
        count_code(metrics["multi_update_candidate_frame_count"]),
        rate_code(metrics["selected_age_nonmissing_rate"]),
        age_code,
        rate_code(stale_rate) if stale_rate is not None else 0,
        *appeared_codes[:4],
        *appeared_codes[4:],
        2 if all(code == 2 for code in appeared_codes) else 1,
        2,
        rate_code(optional_selected / max(1, optional_total)),
        2 if result["FrameCount"].is_unique else 1,
        2 if total_input_rows > total_frames else 1,
        2,
        2,
        2,
        2,
    ]
    return codes, manifest


def output_packet(codes):
    print("MCRS001B1-10")
    for index in range(10):
        block = codes[index * 4 : (index + 1) * 4]
        print(f"A{index + 1}-" + "".join(str(value) for value in block))
    print(f"C-{sum(codes) % 10}")


def zero_packet():
    output_packet([0] * 40)


def main():
    path = choose_input_file()
    if path is None or not path.exists():
        zero_packet()
        return
    columns, _ = read_header(path)
    config = mapping_dialog(columns)
    if not config:
        zero_packet()
        return
    output_dir = Path(__file__).resolve().parent / (
        "multi_camera_reference_selector_v0_1_"
        + datetime.now().strftime("%Y%m%d_%H%M%S")
    )
    codes, _ = build_selector(path, config, output_dir)
    output_packet(codes)


if __name__ == "__main__":
    try:
        main()
    except Exception as exc:
        try:
            Path(__file__).resolve().with_name(
                "multi_camera_reference_selector_v0_1_FATAL.txt"
            ).write_text(f"{type(exc).__name__}: {exc}\n", encoding="utf-8")
        except Exception:
            pass
        zero_packet()
