def normalized(value) -> str:
    return re.sub(r"[^a-z0-9]", "", str(value).lower())

def sniff_csv(path: Path):
    raw = path.read_bytes()[:131072]
    decoded = None
    encoding = None
    for candidate in ["utf-8-sig", "cp932", "shift_jis", "utf-8", "latin1"]:
        try:
            decoded = raw.decode(candidate)
            encoding = candidate
            break
        except UnicodeDecodeError:
            continue
    if decoded is None:
        decoded = raw.decode("latin1", errors="replace")
        encoding = "latin1"
    try:
        separator = csv.Sniffer().sniff(decoded, delimiters=",;\t|").delimiter
    except Exception:
        counts = {item: decoded.count(item) for item in [",", ";", "\t", "|"]}
        separator = max(counts, key=counts.get)
    return encoding, separator


def read_header(path: Path):
    import pandas as pd

    if path.suffix.lower() in {".xlsx", ".xls", ".xlsm"}:
        frame = pd.read_excel(path, nrows=0)
        return list(frame.columns), {"kind": "excel"}
    encoding, separator = sniff_csv(path)
    frame = pd.read_csv(path, sep=separator, encoding=encoding, nrows=0)
    return list(frame.columns), {
        "kind": "csv",
        "encoding": encoding,
        "separator": separator,
    }


def detect_column(columns, include_tokens, exclude_tokens=()):
    candidates = []
    for column in columns:
        key = normalized(column)
        if all(token in key for token in include_tokens) and not any(
            token in key for token in exclude_tokens
        ):
            candidates.append(column)
    return candidates[0] if len(candidates) == 1 else None


def parse_sentinels(text: str):
    values = []
    for item in re.split(r"[,;\s]+", text.strip()):
        if not item:
            continue
        try:
            values.append(float(item))
        except ValueError:
            pass
    return sorted(set(values))


def optional_float(text: str):
    text = text.strip()
    return None if not text else float(text)

def fast_fingerprint(path: Path):
    digest = hashlib.sha256()
    size = path.stat().st_size
    digest.update(str(size).encode("ascii"))
    with path.open("rb") as handle:
        digest.update(handle.read(1024 * 1024))
        if size > 1024 * 1024:
            handle.seek(max(0, size - 1024 * 1024))
            digest.update(handle.read(1024 * 1024))
    return digest.hexdigest()


def read_selected(path: Path, columns, reader_info):
    import pandas as pd

    if reader_info["kind"] == "excel":
        return pd.read_excel(path, usecols=columns)
    return pd.read_csv(
        path,
        sep=reader_info["separator"],
        encoding=reader_info["encoding"],
        usecols=columns,
        low_memory=False,
        memory_map=True,
    )


def valid_number_mask(series, sentinels):
    import numpy as np

    values = series.to_numpy(dtype=float, na_value=np.nan)
    mask = np.isfinite(values)
    for sentinel in sentinels:
        mask &= ~np.isclose(values, sentinel, rtol=1e-12, atol=1e-12, equal_nan=False)
    return mask


def changed(series):
    return series.notna() & series.shift(1).notna() & series.ne(series.shift(1))


def near_one_frame(mask):
    return mask | mask.shift(1, fill_value=False) | mask.shift(-1, fill_value=False)


def rate_code(rate):
    if rate <= 0:
        return 1
    if rate < 0.10:
        return 2
    if rate < 0.90:
        return 3
    return 4


def count_code(count):
    if count == 0:
        return 1
    if count <= 10:
        return 2
    if count <= 100:
        return 3
    return 4


def finite_rate(series):
    return float(series.notna().mean()) if len(series) else 0.0


def boolean_rate(mask, denominator_mask=None):
    if denominator_mask is None:
        return float(mask.mean()) if len(mask) else 0.0
    denominator = int(denominator_mask.sum())
    return float((mask & denominator_mask).sum() / denominator) if denominator else 0.0


def top_values(series, limit=20):
    result = []
    for value, count in series.dropna().value_counts().head(limit).items():
        result.append({"value": float(value), "frame_count": int(count)})
    return result


def safe_number(value):
    if value is None:
        return None
    try:
        value = float(value)
    except (TypeError, ValueError):
        return None
    return value if math.isfinite(value) else None


def profile(path: Path, config: dict, output_dir: Path):
    import numpy as np
    import pandas as pd

    output_dir.mkdir(parents=True, exist_ok=True)
    columns, reader_info = read_header(path)
    role_to_column = {
        key: value
        for key, value in config.items()
        if key
        in {
            "frame",
            "radar_distance",
            "radar_detection",
            "cam1_distance",
            "cam1_lateral",
            "cam1_switching",
            "cam1_counter",
            "cam1_detection_type",
            "cam3_distance",
            "cam3_lateral",
            "cam3_switching",
            "cam3_counter",
            "cam3_detection_type",
            "recognition_id",
            "vehicle_detect",
            "acc_system_status",
            "halt_state",
            "auto_adjust_complete",
        }
        and value is not None
    }
    selected_columns = list(dict.fromkeys(role_to_column.values()))
    before_stat = path.stat()
    started = time.perf_counter()
    source = read_selected(path, selected_columns, reader_info)
    total_rows = len(source)
    source = source.rename(columns={value: key for key, value in role_to_column.items()})
    for role in role_to_column:
        source[role] = pd.to_numeric(source[role], errors="coerce")
    source = source[source["frame"].notna()].copy()
    source["frame"] = source["frame"].astype(float)

    frame_values = source["frame"].to_numpy()
    backward_transitions = int(np.sum(np.diff(frame_values) < 0)) if len(frame_values) > 1 else 0

    conflict_roles = [
        role
        for role in [
            "radar_distance",
            "radar_detection",
            "cam1_distance",
            "cam1_switching",
            "cam1_counter",
            "cam3_distance",
            "cam3_switching",
            "cam3_counter",
            "recognition_id",
        ]
        if role in source.columns
    ]
    grouped = source.groupby("frame", sort=False, dropna=True)
    conflict = grouped[conflict_roles].nunique(dropna=True).gt(1)
    conflict.columns = [f"{column}_intra_frame_conflict" for column in conflict.columns]

    frame = source.drop_duplicates("frame", keep="last").set_index("frame", drop=False)
    frame = frame.join(conflict, how="left")
    for column in conflict.columns:
        frame[column] = frame[column].fillna(False).astype(bool)
    frame = frame.reset_index(drop=True)

    frame["region"] = np.select(
        [
            frame["frame"] < config["valid_start"],
            frame["frame"] < config["switch_start"],
        ],
        ["PRE_VALID", "STABLE"],
        default="SWITCH",
    )
    frame["recognition_selector"] = "UNKNOWN"
    rec_conflict = frame["recognition_id_intra_frame_conflict"]
    match_cam1 = np.isclose(
        frame["recognition_id"], config["recognition_cam1"], rtol=0, atol=1e-12, equal_nan=False
    )
    match_cam3 = np.isclose(
        frame["recognition_id"], config["recognition_cam3"], rtol=0, atol=1e-12, equal_nan=False
    )
    frame.loc[match_cam1 & ~rec_conflict, "recognition_selector"] = "CAM1"
    frame.loc[match_cam3 & ~rec_conflict, "recognition_selector"] = "CAM3"

    previous_selector = frame["recognition_selector"].shift(1)
    mapped_now = frame["recognition_selector"].isin(["CAM1", "CAM3"])
    mapped_previous = previous_selector.isin(["CAM1", "CAM3"])
    frame["recognition_transition"] = (
        mapped_now & mapped_previous & frame["recognition_selector"].ne(previous_selector)
    )
    frame["previous_recognition_selector"] = previous_selector

    frame["cam1_counter_changed"] = changed(frame["cam1_counter"])
    frame["cam3_counter_changed"] = changed(frame["cam3_counter"])
    frame["cam1_switching_changed"] = changed(frame["cam1_switching"])
    frame["cam3_switching_changed"] = changed(frame["cam3_switching"])
    frame["cam1_switching_nonzero"] = frame["cam1_switching"].fillna(0).ne(0)
    frame["cam3_switching_nonzero"] = frame["cam3_switching"].fillna(0).ne(0)

    flag_change_any = frame["cam1_switching_changed"] | frame["cam3_switching_changed"]
    flag_nonzero_any = frame["cam1_switching_nonzero"] | frame["cam3_switching_nonzero"]
    counter_change_any = frame["cam1_counter_changed"] | frame["cam3_counter_changed"]
    frame["switch_flag_change_near"] = near_one_frame(flag_change_any)
    frame["switch_flag_nonzero_near"] = near_one_frame(flag_nonzero_any)
    frame["counter_change_near"] = near_one_frame(counter_change_any)
    frame["selected_counter_changed"] = np.where(
        frame["recognition_selector"].eq("CAM1"),
        frame["cam1_counter_changed"],
        np.where(
            frame["recognition_selector"].eq("CAM3"),
            frame["cam3_counter_changed"],
            False,
        ),
    )

    cam1_distance_valid = pd.Series(
        valid_number_mask(frame["cam1_distance"], config["camera_sentinels"]),
        index=frame.index,
    )
    cam3_distance_valid = pd.Series(
        valid_number_mask(frame["cam3_distance"], config["camera_sentinels"]),
        index=frame.index,
    )
    radar_distance_valid = pd.Series(
        valid_number_mask(frame["radar_distance"], config["radar_sentinels"]),
        index=frame.index,
    )
    radar_detected = np.isclose(
        frame["radar_detection"],
        config["radar_detected_value"],
        rtol=0,
        atol=1e-12,
        equal_nan=False,
    )
    frame["radar_detected"] = radar_detected
    frame["selected_camera_distance"] = np.where(
        frame["recognition_selector"].eq("CAM1") & cam1_distance_valid,
        frame["cam1_distance"],
        np.where(
            frame["recognition_selector"].eq("CAM3") & cam3_distance_valid,
            frame["cam3_distance"],
            np.nan,
        ),
    )
    frame["selected_camera_lateral"] = np.where(
        frame["recognition_selector"].eq("CAM1"),
        frame["cam1_lateral"],
        np.where(
            frame["recognition_selector"].eq("CAM3"),
            frame["cam3_lateral"],
            np.nan,
        ),
    )
    discrepancy_gate = (
        frame["radar_detected"]
        & radar_distance_valid
        & frame["selected_camera_distance"].notna()
    )
    frame["provisional_discrepancy"] = np.where(
        discrepancy_gate,
        frame["radar_distance"] - frame["selected_camera_distance"],
        np.nan,
    )

    transition_fields = [
        "frame",
        "region",
        "previous_recognition_selector",
        "recognition_selector",
        "recognition_id",
        "cam1_switching",
        "cam3_switching",
        "cam1_switching_changed",
        "cam3_switching_changed",
        "switch_flag_change_near",
        "switch_flag_nonzero_near",
        "cam1_counter_changed",
        "cam3_counter_changed",
        "counter_change_near",
        "cam1_distance",
        "cam3_distance",
        "cam1_lateral",
        "cam3_lateral",
        "selected_camera_distance",
        "selected_camera_lateral",
        "radar_distance",
        "radar_detected",
        "provisional_discrepancy",
    ]
    transitions = frame.loc[frame["recognition_transition"], transition_fields].copy()

    pre_mask = frame["region"].eq("PRE_VALID")
    stable_mask = frame["region"].eq("STABLE")
    switch_mask = frame["region"].eq("SWITCH")
    valid_mask = ~pre_mask
    mapped_mask = frame["recognition_selector"].isin(["CAM1", "CAM3"])
    transition_mask = frame["recognition_transition"]

    stable_selector_counts = Counter(frame.loc[stable_mask, "recognition_selector"])
    stable_mapped_total = stable_selector_counts["CAM1"] + stable_selector_counts["CAM3"]
    stable_dominance = (
        max(stable_selector_counts["CAM1"], stable_selector_counts["CAM3"])
        / stable_mapped_total
        if stable_mapped_total
        else 0.0
    )
    switch_selector_values = set(
        frame.loc[switch_mask & mapped_mask, "recognition_selector"].tolist()
    )

    metrics = {
        "total_input_rows": int(total_rows),
        "total_frame_records": int(len(frame)),
        "backward_frame_transitions": backward_transitions,
        "pre_valid_frame_records": int(pre_mask.sum()),
        "stable_frame_records": int(stable_mask.sum()),
        "switch_frame_records": int(switch_mask.sum()),
        "recognition_nonmissing_rate_valid": boolean_rate(
            frame["recognition_id"].notna(), valid_mask
        ),
        "recognition_1_or_3_rate_valid": boolean_rate(mapped_mask, valid_mask),
        "stable_selector_dominance": stable_dominance,
        "recognition_transition_count": int(transition_mask.sum()),
        "transition_flag_change_near_rate": boolean_rate(
            frame["switch_flag_change_near"], transition_mask
        ),
        "transition_flag_nonzero_near_rate": boolean_rate(
            frame["switch_flag_nonzero_near"], transition_mask
        ),
        "transition_counter_change_near_rate": boolean_rate(
            frame["counter_change_near"], transition_mask
        ),
        "recognition_conflict_frame_count": int(rec_conflict.sum()),
        "cam1_switching_nonmissing_rate_valid": boolean_rate(
            frame["cam1_switching"].notna(), valid_mask
        ),
        "cam3_switching_nonmissing_rate_valid": boolean_rate(
            frame["cam3_switching"].notna(), valid_mask
        ),
        "cam1_counter_nonmissing_rate_valid": boolean_rate(
            frame["cam1_counter"].notna(), valid_mask
        ),
        "cam3_counter_nonmissing_rate_valid": boolean_rate(
            frame["cam3_counter"].notna(), valid_mask
        ),
        "cam1_counter_change_count_valid": int((frame["cam1_counter_changed"] & valid_mask).sum()),
        "cam3_counter_change_count_valid": int((frame["cam3_counter_changed"] & valid_mask).sum()),
        "stable_selected_distance_rate": boolean_rate(
            frame["selected_camera_distance"].notna(), stable_mask
        ),
        "switch_selected_distance_rate": boolean_rate(
            frame["selected_camera_distance"].notna(), switch_mask
        ),
        "radar_detected_rate_valid": boolean_rate(frame["radar_detected"], valid_mask),
        "provisional_discrepancy_rate_valid": boolean_rate(
            frame["provisional_discrepancy"].notna(), valid_mask
        ),
        "camera_distance_conflict_frame_count": int(
            frame["cam1_distance_intra_frame_conflict"].sum()
            + frame["cam3_distance_intra_frame_conflict"].sum()
        ),
        "selected_abs_lateral_median": safe_number(
            frame.loc[valid_mask, "selected_camera_lateral"].abs().median()
        ),
        "provisional_discrepancy_median": safe_number(
            frame.loc[valid_mask, "provisional_discrepancy"].median()
        ),
        "provisional_discrepancy_q95_abs": safe_number(
            frame.loc[valid_mask, "provisional_discrepancy"].abs().quantile(0.95)
        ),
    }

    after_stat = path.stat()
    input_unchanged = (
        after_stat.st_size == before_stat.st_size
        and after_stat.st_mtime_ns == before_stat.st_mtime_ns
    )
    frame_path = output_dir / "camera_reference_frame_profile_v0_1.csv"
    transition_path = output_dir / "camera_reference_transition_events_v0_1.csv"
    manifest_path = output_dir / "camera_reference_manifest_v0_1.json"
    log_path = output_dir / "camera_reference_log_v0_1.txt"

    frame.to_csv(frame_path, index=False, encoding="utf-8-sig")
    transitions.to_csv(transition_path, index=False, encoding="utf-8-sig")

    optional_distributions = {}
    for role in [
        "vehicle_detect",
        "acc_system_status",
        "halt_state",
        "auto_adjust_complete",
        "cam1_detection_type",
        "cam3_detection_type",
    ]:
        if role in frame.columns:
            optional_distributions[role] = top_values(frame[role])

    manifest = {
        "version": VERSION,
        "run_id": str(uuid.uuid4()),
        "created_utc": datetime.now(timezone.utc).isoformat(),
        "input": {
            "basename": path.name,
            "size_bytes": before_stat.st_size,
            "fast_fingerprint_sha256": fast_fingerprint(path),
            "input_unchanged": input_unchanged,
        },
        "configuration": {
            key: value for key, value in config.items() if key != "confirmed"
        },
        "metrics": metrics,
        "selector_distributions": {
            "recognition_id": top_values(frame["recognition_id"]),
            "stable_selector": dict(stable_selector_counts),
            "switch_selector_values": sorted(switch_selector_values),
            "cam1_switching_flag": top_values(frame["cam1_switching"]),
            "cam3_switching_flag": top_values(frame["cam3_switching"]),
            "cam1_message_counter": top_values(frame["cam1_counter"]),
            "cam3_message_counter": top_values(frame["cam3_counter"]),
        },
        "optional_signal_distributions": optional_distributions,
        "interpretation_limits": [
            "Recognition_ID=1/3 is a hypothesis under test, not an approved selector rule.",
            "Radar is a provisional reference and has no target ID in this dataset.",
            "Provisional discrepancy is diagnostic only and must not be reported as camera error.",
            "The observed 38239 boundary is a validation label, not an algorithm rule.",
            "MessageCounter changes are used as freshness evidence because decoded values may be forward-filled.",
            "SwitchingFlag value semantics still require DBC/business confirmation.",
        ],
    }
    manifest_path.write_text(
        json.dumps(manifest, ensure_ascii=False, indent=2), encoding="utf-8"
    )
    log_path.write_text(
        "\n".join(
            [
                f"version={VERSION}",
                f"input_basename={path.name}",
                f"elapsed_seconds={time.perf_counter() - started:.3f}",
                f"input_unchanged={input_unchanged}",
                f"metrics={metrics}",
                f"recognition_id_top20={top_values(frame['recognition_id'])}",
                f"stable_selector_counts={dict(stable_selector_counts)}",
                f"switch_selector_values={sorted(switch_selector_values)}",
            ]
        )
        + "\n",
        encoding="utf-8",
    )

    total_frames = len(frame)
    frame_bucket = 1 if total_frames < 50_000 else 2 if total_frames < 200_000 else 3
    stable_code = (
        1
        if stable_mapped_total == 0
        else 4
        if stable_dominance >= 0.95
        else 3
        if stable_dominance >= 0.80
        else 2
    )
    switch_both_code = (
        3
        if {"CAM1", "CAM3"}.issubset(switch_selector_values)
        else 2
        if switch_selector_values
        else 1
    )
    optional_rate = lambda role: (
        rate_code(boolean_rate(frame[role].notna(), valid_mask))
        if role in frame.columns
        else 0
    )
    codes = [
        2,
        2,
        2,
        2 if input_unchanged else 1,
        2 if backward_transitions == 0 else 1,
        frame_bucket,
        2 if pre_mask.any() else 1,
        2 if switch_mask.any() else 1,
        rate_code(metrics["recognition_nonmissing_rate_valid"]),
        rate_code(metrics["recognition_1_or_3_rate_valid"]),
        stable_code,
        switch_both_code,
        count_code(metrics["recognition_transition_count"]),
        rate_code(metrics["transition_flag_change_near_rate"]),
        rate_code(metrics["transition_flag_nonzero_near_rate"]),
        count_code(metrics["recognition_conflict_frame_count"]),
        rate_code(metrics["cam1_switching_nonmissing_rate_valid"]),
        rate_code(metrics["cam3_switching_nonmissing_rate_valid"]),
        rate_code(metrics["cam1_counter_nonmissing_rate_valid"]),
        rate_code(metrics["cam3_counter_nonmissing_rate_valid"]),
        count_code(metrics["cam1_counter_change_count_valid"]),
        count_code(metrics["cam3_counter_change_count_valid"]),
        rate_code(metrics["transition_counter_change_near_rate"]),
        rate_code(metrics["stable_selected_distance_rate"]),
        rate_code(metrics["switch_selected_distance_rate"]),
        rate_code(metrics["radar_detected_rate_valid"]),
        rate_code(metrics["provisional_discrepancy_rate_valid"]),
        count_code(metrics["camera_distance_conflict_frame_count"]),
        optional_rate("halt_state"),
        optional_rate("auto_adjust_complete"),
        2 if pre_mask.any() else 1,
        2,
    ]
    return codes, manifest


def output_packet(codes):
    print("CRL001B1-8")
    for index in range(8):
        block = codes[index * 4 : (index + 1) * 4]
        print(f"A{index + 1}-" + "".join(str(value) for value in block))
    print(f"C-{sum(codes) % 10}")


def zero_packet():
    output_packet([0] * 32)


def main():
    path = choose_input_file()
    if path is None or not path.exists():
        zero_packet()
        return
    columns, _ = read_header(path)
    config = configuration_dialog(columns)
    if not config:
        zero_packet()
        return
    output_dir = Path(__file__).resolve().parent / (
        "camera_reference_logic_profile_v0_1_"
        + datetime.now().strftime("%Y%m%d_%H%M%S")
    )
    codes, _ = profile(path, config, output_dir)
    output_packet(codes)


if __name__ == "__main__":
    try:
        main()
    except Exception as exc:
        try:
            Path(__file__).resolve().with_name(
                "camera_reference_logic_profiler_v0_1_FATAL.txt"
            ).write_text(f"{type(exc).__name__}: {exc}\n", encoding="utf-8")
        except Exception:
            pass
        zero_packet()
